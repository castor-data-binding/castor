51  /**
52   * An abstract factory class for creating GeneralizedFieldHandlers.
53   *
54   * 
55   * @author <a href="kvisco@intalio.com">Keith Visco</a>
56   * @version $Revision$ $Date: 2003-10-23 13:53:59 -0600 (Thu, 23 Oct 2003) $
57   */
58  public abstract class FieldHandlerFactory {
59      
60      /**
61       * Returns an array of the supported Class types
62       * for this FieldHandlerFactory. The array may
63       * be empty, but must not be null.
64       *     
65       * @return an array of supported Class types.
66       */
67      public abstract Class[] getSupportedTypes();
68      
69      /**
70       * Returns true if the given Class type is supported by
71       * this FieldHandlerFactory. If the type is supported,
72       * a call to #createFieldHandler will return a valid
73       * FieldHandler. If the type is not supported, a call
74       * to createFieldHandler may return null or throw
75       * a MappingException.
76       *
77       * @param type the Class type to determine support for.
78       *
79       * @return true if the given Class type is supported.
80       */
81      public abstract boolean isSupportedType(Class type);
82      
83      /**
84       * Creates a GeneralizedFieldHandler for the given class type.
85       * The method should return a new GeneralizedFieldHandler as
86       * an "underlying" FieldHandler will need to be set by the 
87       * caller.
88       *
89       * @param type the Class type to create the FieldHandler for.
90       */
91      public abstract GeneralizedFieldHandler createFieldHandler(Class type) 
92          throws MappingException;
93          
94      
95      
96  } //-- FieldHandlerFactory
