58  /**
59   * A class used to represent an XML node. This is an alternative to DOM which is
60   * too heavy for our purpose (mainly handle XML Fragment when {@literal <any>}
61   * is used in an XML schema). The model is based on XPath Node. An AnyNode can
62   * be a:
63   * <ul>
64   * <li>ELEMENT</li>
65   * <li>ATTRIBUTE</li>
66   * <li>NAMESPACE</li>
67   * <li>COMMENT</li>
68   * <li>TEXT</li>
69   * <li>PROCESSING INSTRUCTION</li>
70   * </ul>
71   *
72   * @author <a href="mailto:arkin@intalio.com">Assaf Arkin</a>
73   * @author <a href="mailto:blandin@intalio.com">Arnaud Blandin</a>
74   * @author <a href="mailto:kvisco@intalio.com">Keith Visco</a>
75   * @version $Revision$ $Date: 2006-04-25 15:08:23 -0600 (Tue, 25 Apr 2006) $
76   */

161     /**
162      * The value of this node defined as follow:
163      * <ul>
164      *     <li>for an element the value is its TEXT NODE value (if any)</li>
165      *     <li>for an attribute the value is the value of the attribute</li>
166      *     <li>for a text node it is the character data</li>
167      *     <li>for a namespace it is the namespace URI that is being bound
168      *     to the namespace prefix</li>
169      *     <li>for a comment it is the content of the comment not including the
170      *     opening &lt;!-- and the closing --&gt;. </li>
171      * </ul>

182     /**
183      * Creates a node given all the necessary information:
184      * type, localName, prefix, uri and value.
185      * This constructor is not user-friendly and launched
186      * RunTime exception is you try to instantiate non-valid node.
187      * @param type the node type.
188      * @param localName the name of the node.
189      * @param prefix the prefix if any of the namespace.
190      * @param uri the namespace uri of this node.
191      * @param value the value of this node.
192      */
193     public AnyNode(short type, String localName, String prefix, String uri, String value) {
194         if ( (type > 6) && (type<1) ) {
195             throw new IllegalArgumentException("Illegal node type");
196         }
197         _nodeType = type;
198 
199         //comment and text nodes don't have name
200         if ( (type>PI) && (localName != null) ) {
201             String err = "This node can not have a local name";
202             throw new IllegalArgumentException(err);
203         }
204         _localName = localName;
205 
206         //for comment, pi or text we should have no namespaces
207         if ((type>NAMESPACE) && ( (uri != null) || (prefix != null)) ) {
208             String err = "This node can not handle namespace";
209             throw new IllegalArgumentException(err);
210         }
211         _uri = uri;
212         _prefix = prefix;
213 
214         //attributes can not be namespaces
215         if (type == AnyNode.ATTRIBUTE)
216             if (localName.startsWith(XMLNS_PREFIX)) {
217                 String err = "Namespaces can't be used as attributes.";
218                 throw new IllegalArgumentException(err);
219             }
220 
221         //you can't set value for element
222         if ( (type == ELEMENT) && (value != null) ) {
223             String err = "You can't set a value for this node type";
224             throw new IllegalArgumentException(err);
225         }
226         _value = value;
227     }

229     /**
230      * Adds an AnyNode to the current node
231      * @param node the node to append
232      */
233     public void addAnyNode(AnyNode node) {
234         if (node == null) {
235             throw new IllegalArgumentException("null argument in addAnyNode");
236         }
237 
238         switch(node.getNodeType()) {
239             case ATTRIBUTE:
240                 addAttribute(node);
241                 break;
242             case NAMESPACE:
243                 addNamespace(node);
244                 break;
245             default:
246                 addChild(node);
247                 break;
248         }
249     }

251     /**
252      * <p>
253      * Adds a child AnyNode to this node. A 'child' can be either an ELEMENT
254      * node, a COMMENT node, a TEXT node or a PROCESSING INSTRUCTION. If the
255      * current node already has a child then the node to add will be append as a
256      * sibling.
257      * <p>
258      * Note: you cannot add a child to a TEXT node.
259      *
260      * @param node
261      *            the node to add.
262      */
263     public void addChild(AnyNode node) {
264         if (node == null) {
265             throw new IllegalArgumentException("null argument in appendChild");
266         }
267 
268         if (node.getNodeType() == ATTRIBUTE || node.getNodeType() == NAMESPACE) {
269             throw new IllegalArgumentException("An Attribute or an Namespace can't be added as a child");
270         }
271 
272         if (this.getNodeType() == TEXT) {
273             throw new IllegalArgumentException("a TEXT node can't have children.");
274         }
275 
276         if (_firstChildNode == null) {
277             _firstChildNode = node;
278         } else if (_firstChildNode.getNodeType() == ATTRIBUTE || _firstChildNode.getNodeType() == NAMESPACE) {
279             _firstChildNode.addChild(node);
280         } else  {
281             _firstChildNode.appendSibling(node);
282         }
283     }

285     /**
286      * Adds an attribute to the current node.
287      * @param node the attribute to add.
288      */
289     public void addAttribute(AnyNode node) {
290         if (node == null) {
291             throw new IllegalArgumentException("null argument in addAttribute");
292         }
293 
294         if (node.getNodeType() != ATTRIBUTE) {
295             throw new IllegalArgumentException("Only attribute can be added as an attribute");
296         }
297 
298         if (_firstChildNode == null) {
299             _firstChildNode = node;
300         } else {
301             if (_firstChildNode.getNodeType() == ATTRIBUTE) {
302                 //if we reach an attribute then we add the node as its sibling
303                 _firstChildNode.appendSibling(node);
304             } else if (_firstChildNode.getNodeType() == NAMESPACE) {
305                 //if we reach an namespace the attributre should be added to
306                 //the first child of the namespace
307                 _firstChildNode.addAttribute(node);
308             } else {
309                 //unplug the current firstNode to add a new one
310                 node.addChild(_firstChildNode);
311                 _firstChildNode = node;
312             }
313         }
314     } //addAttribute
315 
316     /**
317      * Appends an namespace to the current node.
318      * @param node the attribute to add.
319      */
320     public void addNamespace(AnyNode node){
321         if (node == null) {
322             throw new IllegalArgumentException("null argument in addNamespace");
323         }
324 
325         if (node.getNodeType() != NAMESPACE) {
326             throw new IllegalArgumentException("Only namespace can be added as an namespace");
327         }
328 
329         if (_firstChildNode == null) {
330             _firstChildNode = node;
331         } else {
332             if (_firstChildNode.getNodeType() == NAMESPACE) {
333                 //if we reach an namepace then we add the node as its sibling
334                 _firstChildNode.appendSibling(node);
335             } else if (_firstChildNode.getNodeType() == ATTRIBUTE) {
336                 //if we reach an attribute the attributre should be added to
337                 //the first child of the attribute
338                 _firstChildNode.addNamespace(node);
339             } else {
340                 //unplug the current firstNode to add a new one
341                 node.addChild(_firstChildNode);
342                 _firstChildNode = node;
343             }
344         }
345     } //addNamespace

347     /**
348      * Returns the first attribute of the current ELEMENT node
349      * or null.
350      * The next attribute,if any,is the sibling of the returned node.
351      */
352     public AnyNode getFirstAttribute() {
353         if (this.getNodeType() != ELEMENT) {
354             String err = "This node type can not contain attributes";
355             throw new UnsupportedOperationException(err);
356         }
357 
358         boolean found = false;
359         AnyNode tempNode = this.getFirstChildNode();
360         while (tempNode != null && !found) {
361             short type = tempNode.getNodeType();
362             //if the child is not an attribute or a namespace
363             //this element does not have any attribute
364             if (type == ELEMENT || type == COMMENT || type == TEXT || type == PI) {
365                 tempNode = null;
366             } else if (type == NAMESPACE) {
367                 tempNode = tempNode.getFirstChildNode();
368             } else {
369                 found = true;
370             }
371         }
372         return tempNode;
373     }

375     /**
376      * Returns the first namespace of the current ELEMENT node or null. The next
377      * attribute if any is the sibling of the returned node.
378      * @return the first namespace of the current ELEMENT node or null.
379      */
380     public AnyNode getFirstNamespace() {
381         if (this.getNodeType() != ELEMENT) {
382             String err = "This node type can not contain namespaces";
383             throw new UnsupportedOperationException(err);
384         }
385 
386         AnyNode tempNode = this.getFirstChildNode();
387         boolean found = false;
388         while (tempNode != null && !found) {
389             short type = tempNode.getNodeType();
390             //if the child is not an attribute or a namespace
391             //this element does not have any namespace
392             if (type == ELEMENT || type == COMMENT || type == TEXT || type == PI) {
393                 tempNode = null;
394             } else if (type == ATTRIBUTE) {
395                 tempNode = tempNode.getFirstChildNode();
396             } else {
397                 found = true;
398             }
399         }
400         return tempNode;
401     }

403     /**
404      * Returns the first Child node of this node. A 'child' can be either an
405      * ELEMENT node, a COMMENT node, a TEXT node or a PROCESSING INSTRUCTION.
406      *
407      * @return the first child of this node
408      */
409     public AnyNode getFirstChild() {
410         //an ATTRIBUTE or a NAMESPACE can not
411         //have children
412         if (this.getNodeType() == ATTRIBUTE || this.getNodeType() == NAMESPACE) {
413             return null;
414         }
415 
416         //loop througth the first two (in the worst case) nodes
417         //and then return the firstChild if any
418         AnyNode tempNode = this.getFirstChildNode();
419         boolean found = false;
420         while (tempNode != null && !found) {
421             short type = tempNode.getNodeType();
422             if (type == ELEMENT || type == COMMENT || type == TEXT || type == PI) {
423                 found = true;
424             } else if (type == ATTRIBUTE || type == NAMESPACE) {
425                 tempNode = tempNode.getFirstChildNode();
426             }
427         }
428         return tempNode;
429     }

472     /**
473      * Returns the string value of the node. The string value of a text node or
474      * an attribute node is its text value. The string value of an element or a
475      * root node is the concatenation of the string value of all its child
476      * nodes. The string value of a namespace node is its namespace URI. The
477      * string value of a processing instruction is the instruction, and the
478      * string value of a comment is the comment text.
479      *
480      * @return The string value of the node
481      */
482     public String getStringValue() {
483         switch(_nodeType) {
484             case ATTRIBUTE:
485             case TEXT:
486                 return _value;
487             case NAMESPACE:
488                 return _uri;
489                 //not yet supported
490             case PI:
491                 return "";
492             case COMMENT:
493                 return _value;
494             case ELEMENT:
495                 StringBuffer result = new StringBuffer(4096);
496                 AnyNode tempNode = this.getNextSibling();
497                 while (tempNode != null && tempNode.getNodeType() == TEXT) {
498                     result.append(tempNode.getStringValue());
499                     tempNode = tempNode.getNextSibling();
500                 }
501 
502                 tempNode = this.getFirstChild();
503                 while (tempNode != null) {
504                     result.append(tempNode.getStringValue());
505                     tempNode = tempNode.getNextSibling();
506                 }
507                 tempNode = null;
508                 return result.toString();
509             default:
510                 return null;
511         }
512     }

526     /**
527      * Returns the String representation of this AnyNode. The String
528      * representation is a xml well-formed fragment corresponding to the
529      * representation of this node.
530      * @return the String representation of this AnyNode.
531      */
532     public String toString() {
533         Serializer serializer = new BackwardCompatibilityContext().getSerializer();
534         if (serializer == null) {
535             throw new RuntimeException("Unable to obtain serializer");
536         }
537 
538         StringWriter writer = new StringWriter();
539 
540         serializer.setOutputCharStream(writer);
541 
542         try {
543             AnyNode2SAX.fireEvents(this, serializer.asDocumentHandler());
544         } catch (java.io.IOException ioe) {
545             return privateToString();
546         } catch (org.xml.sax.SAXException saxe) {
547             throw new RuntimeException(saxe.getMessage());
548         }
549 
550         return writer.toString();
551     }

553     private String privateToString() {
554         StringBuffer sb = new StringBuffer(4096);
555 
556         if (_elements == null) {
557             _elements = new Stack();
558         }
559 
560         //check the Stack too see if we have
561         //already proceed the node
562         if (_elements.search(this) == -1) {
563             _elements.push(this);
564 
565             if (this.getNodeType() == ELEMENT) {
566                 //open the tag
567                 sb.append("<");
568                 String prefix = getNamespacePrefix();
569                 if (prefix != null) {
570                     sb.append(prefix+":");
571                 }
572                 prefix = null;
573                 sb.append(getLocalName());
574 
575                 //append the attributes
576                 AnyNode tempNode = this.getFirstAttribute();
577                 while (tempNode != null) {
578                     sb.append(" ");
579                     sb.append(tempNode.getLocalName());
580                     sb.append("='"+tempNode.getStringValue()+"'");
581                     tempNode = tempNode.getNextSibling();
582                 }
583 
584                 //append the namespaces
585                 tempNode = this.getFirstNamespace();
586                 while (tempNode != null) {
587                     sb.append(" ");
588                     sb.append(XMLNS_PREFIX);
589                     prefix = tempNode.getNamespacePrefix();
590                     if (prefix != null && prefix.length() != 0) {
591                         sb.append(":"+prefix);
592                     }
593                     sb.append("='"+tempNode.getNamespaceURI()+"'");
594                     tempNode = tempNode.getNextSibling();
595                 }//namespaceNode
596 
597                 tempNode = this.getFirstChild();
598                 if (tempNode != null) {
599                     sb.append(">");
600                     while (tempNode != null) {
601                         sb.append(tempNode.privateToString());
602                         tempNode = tempNode.getNextSibling();
603                     }
604                     //close the tag
605                     sb.append("</"+getLocalName()+">");
606                 } else {
607                     sb.append("/>");
608                 }
609             } else {
610                 sb.append(this.getStringValue());
611             }
612             return sb.toString();
613         }
614         return sb.toString();
615     }//toString()

617     /**
618      * Appends a sibling AnyNode to the current node. The node to append will be
619      * added at the end of the sibling branch.
620      *
621      * @param node
622      *            the node to add
623      */
624     protected void appendSibling(AnyNode node) {
625         if (node == null) {
626             throw new IllegalArgumentException();
627         }
628 
629         if ( ((node.getNodeType() == ATTRIBUTE) || (node.getNodeType()== NAMESPACE)) &&
630               (this.getNodeType() != node.getNodeType())) {
631             String err = "a NAMESPACE or an ATTRIBUTE can only be add as a sibling to a node of the same type";
632             throw new UnsupportedOperationException(err);
633         }
634 
635         if (_nextSiblingNode == null) {
636             //if we already have a TEXT node -> merge
637             if ((node.getNodeType() == TEXT) && (this.getNodeType() == TEXT)) {
638                 mergeTextNode(this,node);
639             } else {
640                 _nextSiblingNode = node;
641             }
642         } else {
643             _nextSiblingNode.appendSibling(node);
644         }
645     }

655     /**
656      * Adds the text value of a TEXT node to another
657      * TEXT node.
658      * @param node1 the AnyNode that receives the text value
659      * @param node2 the AnyNode that needs to be merges with node1.
660      */
661     private void mergeTextNode(AnyNode node1, AnyNode node2) {
662         if (node1.getNodeType() != node2.getNodeType()) {
663             return;
664         }
665         if (node1.getNodeType() != AnyNode.TEXT) {
666             return;
667         }
668 
669         StringBuffer temp = new StringBuffer(node1.getStringValue());
670         temp.append(node2.getStringValue());
671         node1._value = temp.toString();
672         node2 = null;
673     }
