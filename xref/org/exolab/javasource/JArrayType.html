14  package org.exolab.javasource;
15  
16  /**
17   * JType sub-class for Arrays.
18   * 
19   * @author <a href="mailto:werner DOT guttmann AT gmx DOT net">Werner Guttman</a>
20   * @version $Revision: 5951 $ $Date: 2006-04-25 16:09:10 -0600 (Tue, 25 Apr 2006) $
21   * @since 1.0.4
22   */
23  public final class JArrayType extends JComponentizedType {
24    // --------------------------------------------------------------------------
25  
26    /**
27     * Creates an instance of a array type, of type 'name'.
28     * 
29     * @param componentType Component type.
30     * @param useJava50 True if Java 5.0 should be generated.
31     */
32    public JArrayType(final JType componentType, final boolean useJava50) {
33      super(componentType.getName(), componentType, useJava50);
34    }
35  
36    // --------------------------------------------------------------------------
37  
38    /**
39     * Returns the String representation of this JType, which is simply the name of this type.
40     * 
41     * @return The String representation of this JType.
42     */
43    public String toString() {
44      return getComponentType().toString() + "[]";
45    }
46  
47    // --------------------------------------------------------------------------
48  }
