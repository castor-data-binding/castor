68  /**
69   * This class encapsulates all the logic to run the test patterns for the Castor
70   * marshalling framework. This include introspection and mapping.

72   * @author <a href="mailto:gignoux@kernelcenter.com">Sebastien Gignoux</a>
73   * @author <a href="mailto:blandin@intalio.com">Arnaud Blandin</a>
74   * @version $Revision: 6787 $ $Date: 2004-09-10 18:23:03 -0600 (Fri, 10 Sep 2004) $

76  public class MarshallingFrameworkTestCase extends XMLTestCase {

78      /**
79       * Contains the configuration for this test case. The configuration is
80       * directly read for the test descriptor file located in a jar or in a
81       * directory.
82       */
83      protected final MarshallingTest _marshallingConf;
84      /** If true, the randomize() function has been implemented in the root class. */
85      protected final boolean         _hasRandom;
86  
87      /**
88       * Creates a CTF test case for the Marshalling framework.
89       *
90       * @param test A Test Case
91       * @param unit A configuration element for a test case from a TestDescriptor
92       *            configuration file
93       * @param marshalling a Marshalling test definition from a TestDescriptor
94       *            configuration file
95       */
96      public MarshallingFrameworkTestCase(final CastorTestCase test, final UnitTestCase unit,
97                                          final MarshallingTest marshalling) {
98          super(test, unit);
99          _marshallingConf = marshalling;
100 
101         RootType rootType = _marshallingConf.getRoot_Object();
102         if (rootType == null) {
103             throw new IllegalArgumentException("You must give a root object for a Marshaling Test: "
104                     + _outputRootFile + ", " +  getName());
105         }
106 
107         _rootClassName = rootType.getContent();
108         if (_rootClassName == null) {
109             throw new IllegalArgumentException("You must give a root object for a Marshaling Test:"
110                     + _outputRootFile + ", " +  getName());
111         }
112 
113         _hasRandom = rootType.getRandom();
114         _hasDump   = rootType.getDump();
115     }
116 
117     /**
118      * Create a new MarshallingFrameworkTestCase with the given name and a null
119      * marshalling configuration.  This constructor should not be used!
120      *
121      * @param name Name for the MarshallingFrameworkTestCase
122      */
123     public MarshallingFrameworkTestCase(final String name) {
124         super(name);
125         throw new IllegalArgumentException("You cannot use the name-only constructor");
126     }
127 
128     /**
129      * Returns the test suite for this given test setup.
130      * @return the test suite for this given test setup.
131      */
132     public Test suite() {
133         TestSuite suite = new TestSuite(_name);
134 
135         String name = getTestSuiteName();
136         name = (name != null) ? name + "#" + _name : _name;
137 
138         if (_unitTest.getCustomTest() != null) {
139             suite.addTest(new TestWithCustomTest(name, this));
140         } else {
141             suite.addTest(new TestWithReferenceDocument(name, this));
142             if (_hasRandom) {
143                 suite.addTest(new TestWithRandomObject(name, this));
144             }
145         }
146         return suite;
147     }
148 
149     /**
150      * Sets up this test suite. Loads the mapping file if any.
151      * <p>
152      * Nothing in this setUp() method should ever be expected to fail. Thus,
153      * there are no checks against _failure. If anything goes wrong here, there
154      * is a problem with the individual test case configuration.
155      *
156      * @throws Exception
157      *             if anything goes wrong
158      */
159     protected void setUp() throws java.lang.Exception {
160         verbose("\n================================================");
161         verbose("Test suite '" + _test.getName() + "': setting up test '" + _name + "'");
162         verbose("================================================\n");
163 
164         FileServices.copySupportFiles(_test.getTestFile(), _outputRootFile);
165 
166         // Compile the source directory for this test, if not already done
167         if (!_test.isDirectoryCompiled()) {
168             verbose("-->Compiling any necessary source files in " + _outputRootFile);
169             Compiler compiler = CompilerFactory.createInstance(_outputRootFile);
170             if (_unitTest.hasJavaSourceVersion()) {
171                 compiler.setJavaSourceVersion(_unitTest.getJavaSourceVersion());
172             }
173             try {
174                 compiler.compileDirectory();
175                 _test.setDirectoryCompiled(true);
176             } catch (CompilationException e) {
177                 if (_printStack) {
178                     e.printStackTrace(System.out);
179                 }
180                 fail("Build Failed: " + e.getMessage());
181             }
182         }
183 
184         //-- Add outputRoot to classpath
185         ClassLoader loader = _test.getClassLoader();
186         loader = new URLClassLoader(new URL[] {_outputRootFile.toURL()}, loader);
187         _test.setClassLoader(loader);
188         getXMLContext().getInternalContext().setClassLoader(loader);
189 
190         verbose("Root class specified in TestDescriptor...");
191         verbose("Loading class: " + _rootClassName);
192         _rootClass = loader.loadClass(_rootClassName);
193 
194         // Try to load the mapping file if any, else we will use the introspector
195         String mappingFilePath = null;
196         if (_unitTest.getUnitTestCaseChoice() != null) {
197             mappingFilePath = _unitTest.getUnitTestCaseChoice().getMapping_File();
198         }
199 
200         if (mappingFilePath != null) {
201             configureMapping(loader, mappingFilePath);
202         } else {
203             verbose("##### TESTING INTROSPECTION #####");
204             _mapping = null;
205         }
206     }
207 
208     /**
209      * Clean up after a test -- nothing to do except display output.
210      * @throws Exception if anything goes wrong
211      */
212     protected void tearDown() throws java.lang.Exception {
213         verbose("\n================================================");
214         verbose("Test suite '" + _test.getName() + "': test '" + _name + "' complete.");
215         verbose("================================================\n");
216     }

218     /**
219      * For a test case with a mapping, load the mapping. If there is a listener,
220      * initialize it.
221      *
222      * @param loader ClassLoader for this test case
223      * @param mappingFilePath Path to the mapping file
224      * @throws Exception if anything goes wrong during the test
225      */
226     private void configureMapping(final ClassLoader loader, final String mappingFilePath) throws Exception {
227         verbose("##### TESTING MAPPING #####");
228         verbose("Mapping file: " + mappingFilePath);
229         InputStream mappingFile = loader.getResourceAsStream(mappingFilePath);
230 
231         if (mappingFile == null) {
232             throw new FileNotFoundException("Unable to locate the mapping file '"
233                      + mappingFilePath + "' for the test '" + _test.getName() + "'");
234         }
235 
236         _mapping = new Mapping(loader);
237         InputSource source = new InputSource(mappingFile);
238         source.setSystemId(mappingFilePath);
239         _mapping.loadMapping(source);
240 
241         ListenerType listener = _unitTest.getListener();
242         if (listener != null) {
243             String listenerName = listener.getClassName();
244             try {
245                 // See if we can load the class...
246                 initializeListeners(listener);
247             } catch (ClassNotFoundException cnfex) {
248                 //Class#forName
249                 fail("The listener '" + listenerName + "' cannot be found in the CLASSPATH");
250             } catch (InstantiationException iex) {
251                 fail("The listener '" + listenerName + "' cannot be instantiated");
252             } catch (IllegalAccessException iaex) {
253                 fail("Constructing a '" + listenerName + "' failed: " + iaex);
254             }
255             verbose("##### TESTING LISTENER CLASS " + listenerName + " #####");
256         } // listener != null;
257     }
