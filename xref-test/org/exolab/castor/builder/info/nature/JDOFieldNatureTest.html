15  package org.exolab.castor.builder.info.nature;
16  
17  import junit.framework.TestCase;
18  
19  import org.exolab.castor.builder.factory.FieldInfoFactory;
20  import org.exolab.castor.builder.info.FieldInfo;
21  import org.exolab.castor.builder.types.XSClass;
22  import org.exolab.javasource.JClass;
23  
24  /**
25   * Tests access to {@link FieldInfo} properties via {@link JDOFieldInfoNature}. Remember that
26   * behavior of properties not set before is specified in the {@link BaseNature} class and tested in
27   * {@link BaseNatureTest}. Property implementation will not be tested in here.
28   * 
29   * @author Lukas Lang
30   * @since 1.2.1
31   * 
32   */
33  public final class JDOFieldNatureTest extends TestCase {
34  
35    /**
36     * Tests get and set of column name.
37     */
38    public void testColumnName() {
39      FieldInfoFactory factory = new FieldInfoFactory();
40      FieldInfo isbn = factory.createFieldInfo(new XSClass(new JClass("Book")), "isbn");
41      isbn.addNature(JDOFieldInfoNature.class.getName());
42      JDOFieldInfoNature jdo = new JDOFieldInfoNature(isbn);
43  
44      jdo.setColumnName("ISBN");
45      assertEquals("ISBN", jdo.getColumnName());
46  
47    }
48  
49    /**
50     * Tests get and set of column type.
51     */
52    public void testColumnType() {
53      FieldInfoFactory factory = new FieldInfoFactory();
54      FieldInfo isbn = factory.createFieldInfo(new XSClass(new JClass("Book")), "isbn");
55      isbn.addNature(JDOFieldInfoNature.class.getName());
56      JDOFieldInfoNature jdo = new JDOFieldInfoNature(isbn);
57  
58      jdo.setColumnType("varchar");
59      assertEquals("varchar", jdo.getColumnType());
60    }
61  
62    /**
63     * Tests get and set of read only.
64     */
65    public void testReadOnly() {
66      FieldInfoFactory factory = new FieldInfoFactory();
67      FieldInfo isbn = factory.createFieldInfo(new XSClass(new JClass("Book")), "isbn");
68      isbn.addNature(JDOFieldInfoNature.class.getName());
69      JDOFieldInfoNature jdo = new JDOFieldInfoNature(isbn);
70      jdo.setReadOnly(true);
71      assertEquals(true, jdo.isReadOnly());
72    }
73  
74    /**
75     * Tests get and set of dirty.
76     */
77    public void testDirty() {
78      FieldInfoFactory factory = new FieldInfoFactory();
79      FieldInfo isbn = factory.createFieldInfo(new XSClass(new JClass("Book")), "isbn");
80      isbn.addNature(JDOFieldInfoNature.class.getName());
81      JDOFieldInfoNature jdo = new JDOFieldInfoNature(isbn);
82      jdo.setDirty(true);
83      assertEquals(true, jdo.isDirty());
84    }
85  
86  }
