16  package org.castor.core.annotationprocessing;
17  
18  import java.lang.annotation.*;
19  
20  import junit.framework.TestCase;
21  
22  import org.castor.core.nature.BaseNature;
23  import org.castor.core.nature.PropertyHolder;
24  import org.castor.core.annotationprocessing.AnnotationProcessor;
25  import org.castor.core.annotationprocessing.BaseAnnotationProcessingService;
26  
27  public class BaseAnnotationProcessingServiceTest extends TestCase {
28  
29      @Retention(RetentionPolicy.RUNTIME)
30      public @interface SupportedAnnotation {
31      }
32  
33      @Retention(RetentionPolicy.RUNTIME)
34      public @interface UnsupportedAnnotation {
35      }
36  
37      /**
38       * This is a {@link AnnotationProcessor} that actually does nothing, except
39       * telling that it processed the given Annotation if it was of Type
40       * {@link SupportedAnnotation}. It's just a dummy.
41       */
42      class SupportedAnnotationProcessor implements AnnotationProcessor {
43  
44          public Class<? extends Annotation> forAnnotationClass() {
45              return SupportedAnnotation.class;
46          }
47  
48          public <I extends BaseNature, A extends Annotation> boolean processAnnotation(
49                  I info, A annotation) {
50              if (annotation instanceof SupportedAnnotation) {
51                  // do
52                  return true;
53              }
54              return false;
55          }
56      }
57  
58      /**
59       * A simple empty class, that just has 2 Annotations - for one (
60       * {@link SupportedAnnotation}) we have a dummy processor, for the other one
61       * ( {@link UnsupportedAnnotation}) we don't have a processor.
62       * 
63       */
64      @SupportedAnnotation
65      @UnsupportedAnnotation
66      class AnnotationHolder {
67  
68      }
69  
70      /**
71       * A dummy property holder needed for use with {@link BaseNature}s. In real
72       * usage within Castor, this would be a ClassInfo or FieldInfo.
73       * 
74       */
75      class MyPropertyHolder implements PropertyHolder {
76  
77          public Object getProperty(String arg0) {
78              return null;
79          }
80  
81          public void setProperty(String arg0, Object arg1) {
82          }
83  
84          public void addNature(String arg0) {
85          }
86  
87          public boolean hasNature(String arg0) {
88              return true;
89          }
90  
91      }
92  
93      @Override
94      protected void setUp() throws Exception {
95          assertTrue(AnnotationHolder.class
96                  .isAnnotationPresent(SupportedAnnotation.class));
97          assertTrue(AnnotationHolder.class
98                  .isAnnotationPresent(UnsupportedAnnotation.class));
99  
100     }
101 
102     public void testAddGetAnnotationProcessor() {
103         BaseAnnotationProcessingService baps = new BaseAnnotationProcessingService();
104         SupportedAnnotationProcessor annotationProcessor = new SupportedAnnotationProcessor();
105 
106         assertNotNull(baps.getAnnotationProcessors());
107         assertFalse(baps.getAnnotationProcessors()
108                 .contains(annotationProcessor));
109 
110         baps.addAnnotationProcessor(annotationProcessor);
111 
112         assertNotNull(baps.getAnnotationProcessors());
113         assertTrue(baps.getAnnotationProcessors().contains(annotationProcessor));
114 
115     }
116 
117     public void testProcessAnnotations() {
118         BaseAnnotationProcessingService baps = new BaseAnnotationProcessingService();
119         SupportedAnnotationProcessor annotationProcessor = new SupportedAnnotationProcessor();
120         baps.addAnnotationProcessor(annotationProcessor);
121         Annotation[] annotations = AnnotationHolder.class.getAnnotations();
122         assertEquals(2, annotations.length);
123         /*
124          * now annotations[] contains @SupportedAnnotation and
125          * @UnsupportedAnnotation
126          */
127 
128         MyPropertyHolder holder = new MyPropertyHolder();
129         holder.addNature(BaseNature.class.getName());
130 
131         BaseNature info = new BaseNature(holder) {
132             public String getId() {
133                 return BaseNature.class.getName();
134             }
135         };
136 
137         /*
138          * try to process all annotations
139          */
140         Annotation[] unprocessed = baps.processAnnotations(info, annotations);
141 
142         /*
143          * we know, that we don't have processor for @UnsupportedAnnotation, so
144          * 
145          * @UnsupportedAnnotation is put into unprocessed[]
146          */
147         assertNotNull(unprocessed);
148         assertEquals(1, unprocessed.length);
149         assertEquals(UnsupportedAnnotation.class, unprocessed[0]
150                 .annotationType());
151     }
152 
153     public void testProcessAnnotation() {
154         BaseAnnotationProcessingService baps = new BaseAnnotationProcessingService();
155         baps.addAnnotationProcessor(new SupportedAnnotationProcessor());
156 
157         MyPropertyHolder holder = new MyPropertyHolder();
158         holder.addNature(BaseNature.class.getName());
159         BaseNature info = new BaseNature(holder) {
160             public String getId() {
161                 return BaseNature.class.getName();
162             }
163         };
164 
165         /*
166          * our AnnotationHolder has 2 Annotations: @UnsupportedAnnotation and
167          * 
168          * @SupportedAnnotation, so we get them
169          */
170         Annotation unsupportedAnnotation = AnnotationHolder.class
171                 .getAnnotation(UnsupportedAnnotation.class);
172         Annotation supportedAnnotation = AnnotationHolder.class
173                 .getAnnotation(SupportedAnnotation.class);
174         assertNotNull(unsupportedAnnotation);
175         assertNotNull(supportedAnnotation);
176 
177         /*
178          * we know, that we don't have processor for @UnsupportedAnnotation, so
179          * processing it will return false.
180          */
181         assertFalse(baps.processAnnotation(info, unsupportedAnnotation));
182         assertTrue(baps.processAnnotation(info, supportedAnnotation));
183 
184     }
